function Avatar({ initials, hue="slate", size=40 }) {
  const h = HUE[hue] || HUE.slate;
  return <div style={{ width:size, height:size, borderRadius:size*0.3, background:h.bg, color:h.text,
    display:"flex", alignItems:"center", justifyContent:"center",
    fontWeight:800, fontSize:size*0.3, fontFamily:"'Plus Jakarta Sans',sans-serif",
    flexShrink:0, border:`1.5px solid ${h.text}22`, letterSpacing:"-0.01em" }}>{initials}</div>;
}

function Badge({ children, color=T.indigo, small=false }) {
  return <span style={{ display:"inline-flex", alignItems:"center", gap:3,
    padding: small ? "1px 6px" : "2px 9px", borderRadius:6,
    background: color+"18", color, fontSize: small?8:9,
    fontWeight:800, letterSpacing:"0.07em", textTransform:"uppercase",
    border:`1px solid ${color}22`, whiteSpace:"nowrap",
    fontFamily:"'Outfit',sans-serif" }}>{children}</span>;
}

function Pill({ children, active, onClick, color=T.purple }) {
  return <button onClick={onClick} style={{
    padding:"6px 14px", borderRadius:20, border:`1.5px solid ${active ? color : T.border}`,
    background: active ? color+"12" : T.surface,
    color: active ? color : T.slate500,
    fontSize:11, fontWeight:700, cursor:"pointer", whiteSpace:"nowrap",
    transition:"all .15s", fontFamily:"'Outfit',sans-serif" }}>{children}</button>;
}

function Bar({ value, color=T.indigo, h=5, bg="rgba(226,232,240,.8)" }) {
  return <div style={{ width:"100%", height:h, borderRadius:h, background:bg, overflow:"hidden" }}>
    <div style={{ width:`${Math.min(value,100)}%`, height:"100%", borderRadius:h, background:color, transition:"width .6s cubic-bezier(.4,0,.2,1)" }}/>
  </div>;
}

function Card({ children, style={}, onClick }) {
  const [press, setPress] = useState(false);
  return <div onClick={onClick}
    onPointerDown={()=>onClick&&setPress(true)}
    onPointerUp={()=>setPress(false)} onPointerLeave={()=>setPress(false)}
    style={{ background:T.surface, borderRadius:20, border:`1px solid ${T.border}`,
      boxShadow:"0 1px 8px rgba(0,0,0,.05)", transition:"all .15s",
      transform: press?"scale(.98)":"scale(1)", cursor:onClick?"pointer":"default", ...style }}>
    {children}
  </div>;
}

function GlassBtn({ children, onClick, primary=false, danger=false, style={}, disabled=false }) {
  const [press, setPress] = useState(false);
  const bg    = primary ? T.purple : danger ? "#fff0f2" : T.surface;
  const clr   = primary ? "#fff"   : danger ? T.rose    : T.slate700;
  const bdr   = primary ? "none"   : danger ? `1.5px solid ${T.rose}30` : `1.5px solid ${T.border}`;
  const shad  = primary ? `0 4px 18px ${T.purple}40` : "0 1px 4px rgba(0,0,0,.06)";
  return <button onClick={onClick} disabled={disabled}
    onPointerDown={()=>setPress(true)} onPointerUp={()=>setPress(false)} onPointerLeave={()=>setPress(false)}
    style={{ background:bg, color:clr, border:bdr, borderRadius:12,
      padding:"10px 18px", fontSize:13, fontWeight:700, cursor:disabled?"not-allowed":"pointer",
      display:"flex", alignItems:"center", justifyContent:"center", gap:6,
      transform:press?"scale(.97)":"scale(1)", boxShadow:shad,
      transition:"all .15s", fontFamily:"'Outfit',sans-serif",
      opacity:disabled?.5:1, ...style }}>{children}</button>;
}

function SegTab({ options, active, onChange }) {
  return <div style={{ display:"flex", background:"rgba(226,232,240,.5)", borderRadius:14, padding:3 }}>
    {options.map(({id,label})=><button key={id} onClick={()=>onChange(id)} style={{
      flex:1, padding:"8px 0", borderRadius:10, border:"none",
      background: active===id ? T.surface : "transparent",
      color: active===id ? T.slate800 : T.slate400,
      fontSize:11, fontWeight:700, cursor:"pointer",
      boxShadow: active===id ? "0 1px 6px rgba(0,0,0,.08)" : "none",
      transition:"all .18s", fontFamily:"'Outfit',sans-serif" }}>{label}</button>)}
  </div>;
}

function useToast() {
  const [toasts, setToasts] = useState([]);
  const addToast = useCallback((msg, type="info") => {
    const id = Date.now();
    setToasts(t => [...t, { id, msg, type }]);
    setTimeout(() => setToasts(t => t.filter(x => x.id !== id)), 3500);
  }, []);
  const removeToast = useCallback(id => setToasts(t => t.filter(x => x.id !== id)), []);
  return { toasts, addToast, removeToast };
}

        <input value={form.value} onChange={e=>setForm({...form,value:e.target.value})} placeholder="Hodnota (např. 4.2M Kč)" style={{ padding:"11px 12px", borderRadius:12, border:`1.5px solid ${T.border}` }}/>
        <select value={form.stageId} onChange={e=>setForm({...form,stageId:e.target.value})} style={{ padding:"11px 12px", borderRadius:12, border:`1.5px solid ${T.border}` }}>
          {STAGES_DATA.map(s=><option key={s.id} value={s.id}>{s.title}</option>)}
        </select>
      </div>
      <div style={{ display:"flex", gap:8, marginTop:10 }}>
        <GlassBtn style={{ flex:1 }} onClick={onClose}>Zrušit</GlassBtn>
        <GlassBtn primary style={{ flex:1 }} onClick={()=>{ addToast("Obchod vytvořen","success"); onClose(); }}>Vytvořit</GlassBtn>
      </div>
    </Sheet>
  );
}

// ═══════════════════════════════════════════════
// APP ROOT
// ═══════════════════════════════════════════════
export default function AidvisoraApp() {
  const [loggedIn, setLoggedIn] = useState(false);
  const [activeTab, setActiveTab] = useState("home");
  const [subStack, setSubStack] = useState([]);
  const [role, setRole] = useState(ROLES.ADVISOR);
  const [notifOpen, setNotifOpen] = useState(false);
  const [quickOpen, setQuickOpen] = useState(false);
  const [confirm, setConfirm] = useState(null);
  const [taskWizardOpen, setTaskWizardOpen] = useState(false);
  const [requestWizardOpen, setRequestWizardOpen] = useState(false);
  const [documentUploadOpen, setDocumentUploadOpen] = useState(false);
  const [newDealWizardOpen, setNewDealWizardOpen] = useState(false);
  const [notifData, setNotifData] = useState(NOTIFICATIONS_DATA);
  const { toasts, addToast, removeToast } = useToast();

  const currentSub = subStack[subStack.length-1] || null;
  const unread = notifData.filter(n=>!n.read).length;

  const navigate = (id, params={}) => {
    if (NAV_ITEMS.some(n=>n.id===id)) {
      setActiveTab(id);
      setSubStack([]);
      return;
    }
    if (id==="request_wizard") { setRequestWizardOpen(true); return; }
    if (id==="document_upload") { setDocumentUploadOpen(true); return; }
    if (id==="new_deal_wizard") { setNewDealWizardOpen(true); return; }
    setSubStack(prev=>[...prev,{ id, params }]);
  };
  const goBack = () => setSubStack(prev=>prev.slice(0,-1));

  const markRead = id => setNotifData(prev=>prev.map(n=>n.id===id?{...n,read:true}:n));
  const markAll = () => setNotifData(prev=>prev.map(n=>({...n,read:true})));

  const renderRoot = () => {
    if (activeTab==="home") return <DashboardScreen navigate={navigate} role={role}/>;
    if (activeTab==="tasks") return <TasksScreen openWizard={()=>setTaskWizardOpen(true)} addToast={addToast}/>;
    if (activeTab==="clients") return <ClientsScreen navigate={navigate}/>;
    if (activeTab==="pipeline") return <PipelineScreen navigate={navigate}/>;
    if (activeTab==="menu") return <MenuScreen navigate={navigate} role={role}/>;
    return null;
  };

  const renderSub = () => {
    if (!currentSub) return null;
    const id = currentSub.id;
    if (id==="requests") return <RequestsScreen navigate={navigate} addToast={addToast} openRequestWizard={()=>setRequestWizardOpen(true)}/>;
    if (id==="documents") return <DocumentsScreen navigate={navigate} addToast={addToast} openDocumentUpload={()=>setDocumentUploadOpen(true)}/>;
    if (id==="deal_detail") return <DealDetailScreen deal={currentSub.params?.deal} navigate={navigate} addToast={addToast}/>;
    if (id==="product_detail") return <ProductDetailScreen product={currentSub.params?.product} navigate={navigate} addToast={addToast}/>;
    if (id==="settings") return <SettingsScreen role={role} setRole={setRole} addToast={addToast}/>;
    if (id==="client_detail") return <ClientDetailScreen navigate={navigate} addToast={addToast} client={currentSub.params?.client}/>;
    if (id==="household") return <HouseholdScreen/>;
    if (id==="ai_assistant") return <AIAssistantScreen addToast={addToast}/>;
    if (id==="analyses") return <AnalysesScreen addToast={addToast}/>;
    if (id==="calcs") return <CalculatorsScreen addToast={addToast}/>;
    if (id==="plan") return <PlanScreen/>;
    if (id==="team") return <TeamScreen addToast={addToast}/>;
    if (id==="calendar") return <CalendarScreen addToast={addToast}/>;
    if (id==="production") return <ProductionScreen/>;
    if (id==="mindmap") return <MindmapScreen/>;
    if (id==="messages") return <MessagesScreen navigate={navigate}/>;
    if (id==="messages_chat") return <ChatScreen/>;
    return <EmptyState icon={Ico.Warn} title="Neznámá obrazovka" body={`Subview "${id}" není registrovaný.`}/>;
  };

  const fabVisible = (() => {
    if (!loggedIn) return false;
    if (!currentSub) return ["home","tasks","clients","pipeline"].includes(activeTab);
    return ["requests","documents"].includes(currentSub.id);
  })();

  const onFabOpen = () => {
    if (!currentSub && (activeTab==="home"||activeTab==="tasks")) return setTaskWizardOpen(true);
    if (!currentSub && activeTab==="pipeline") return setNewDealWizardOpen(true);
    if (currentSub?.id==="requests") return setRequestWizardOpen(true);
    if (currentSub?.id==="documents") return setDocumentUploadOpen(true);
    setQuickOpen(true);
  };

  return (
    <div style={{ minHeight:"100vh", background:`linear-gradient(135deg,${T.navyDark} 0%,#1a0d3a 50%,${T.navy} 100%)`, display:"flex", alignItems:"center", justifyContent:"center", padding:24, fontFamily:"'Outfit',sans-serif" }}>
      <style>{GLOBAL_CSS}</style>
      <div style={{ width:390, height:844, background:T.bg, borderRadius:52, border:"10px solid #050608",
        boxShadow:"0 40px 120px rgba(0,0,0,.65), inset 0 0 0 1px rgba(255,255,255,.04)",
        display:"flex", flexDirection:"column", overflow:"hidden", position:"relative" }}>
        <StatusBar dark={!loggedIn}/>
        <Toast toasts={toasts} removeToast={removeToast}/>

        {!loggedIn ? (
          <div style={{ flex:1, overflow:"hidden" }}>
            <LoginScreen onLogin={()=>setLoggedIn(true)}/>
          </div>
        ) : (
          <>
            <AppHeader subView={!!currentSub} onBack={goBack} subLabel={SUB_LABELS[currentSub?.id]} notifBadge={unread} onNotif={()=>setNotifOpen(true)}/>
            <div key={(currentSub?.id||activeTab)+(subStack.length)} className="slide-up" style={{ flex:1, overflowY:"auto", padding:"16px 14px 0", position:"relative" }}>
              {currentSub ? renderSub() : renderRoot()}
            </div>
            {!currentSub && <BottomNav active={activeTab} onChange={id=>{ setActiveTab(id); setSubStack([]); }}/>}
            {fabVisible && <FAB onOpen={onFabOpen}/>}
          </>
        )}

        <NotificationsSheet open={notifOpen} onClose={()=>setNotifOpen(false)} notifications={notifData} onMarkRead={markRead} onMarkAll={markAll} navigate={navigate}/>
        <QuickActionsSheet open={quickOpen} onClose={()=>setQuickOpen(false)} navigate={navigate}
          openTaskWizard={()=>setTaskWizardOpen(true)}
          openRequestWizard={()=>setRequestWizardOpen(true)}
          addToast={addToast}/>
        <TaskWizard open={taskWizardOpen} onClose={()=>setTaskWizardOpen(false)} addToast={addToast}/>
        <RequestWizard open={requestWizardOpen} onClose={()=>setRequestWizardOpen(false)} addToast={addToast}/>
        <DocumentUploadSheet open={documentUploadOpen} onClose={()=>setDocumentUploadOpen(false)} addToast={addToast}/>
        <NewDealWizard open={newDealWizardOpen} onClose={()=>setNewDealWizardOpen(false)} addToast={addToast}/>
        <ConfirmDialog open={!!confirm} title={confirm?.title||""} body={confirm?.body||""}
          onCancel={()=>setConfirm(null)} onConfirm={()=>{ confirm?.onConfirm?.(); setConfirm(null); }} danger={!!confirm?.danger}/>

        <div style={{ position:"absolute", bottom:7, left:"50%", transform:"translateX(-50%)", width:100, height:4, borderRadius:2, background:"rgba(0,0,0,.18)", pointerEvents:"none", zIndex:200 }}/>
      </div>
    </div>
  );
}
